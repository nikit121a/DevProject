                                  To implement the functionality of saving and restoring the layout, modify the code of the GridViewToolbarCreater class as shown below:

                                  1) Modify the code of the FillGridViewToolbarWithItems method to add saveLayouGridViewToolbarItem GridViewToolbarItem into GridViewToolbar:

                                  [ASPx]
                                  private static GridViewToolbar FillGridViewToolbarWithItems(ASPxGridView ASPxGridView, string popupControlClientInstanceName)
{
    ...
    GridViewToolbarItem saveLayouGridViewToolbarItem = CreateGridViewToolbarItemWithCommand(GridViewToolbarCommand.Custom);
     saveLayouGridViewToolbarItem.Name = "saveLayouGridViewToolbarItem";
    saveLayouGridViewToolbarItem.Text = "Save ASPxGridView's layout";
     gridViewToolbar.Items.Add(saveLayouGridViewToolbarItem);
    ...
}

                                  2) Modify the code of the RecreateGridViewToolbar method to add client-side event handlers for processing the following client-side actions:

                                  - adding a new layout row into ASPxGridView, 
                                  - clicking on the load or cancel buttons, 
                                  - showing or hiding ASPxPopupControl.

                                  These client-side actions will be performed based on the properties of received parameters: ASPxGridView and ASPxPopupControl, in which the layout database rows will be displayed:

                                  [C#]
                                  public static void RecreateGridViewToolbar(ASPxGridView ASPxGridView, ASPxPopupControl ASPxPopupControl)
{
    ASPxGridView popupASPxGridView = ASPxPopupControl.FindControl("popupASPxGridView") as ASPxGridView;
    string popupASPxGriViewClientInstanceName = popupASPxGridView.ClientInstanceName;
    string popupControlClienInstanceName = ASPxPopupControl.ClientInstanceName;
     if (!(ASPxGridView.Page.IsPostBack))
    {
        ASPxGridView.Toolbars.Add(FillGridViewToolbarWithItems(ASPxGridView, "ASPxPopupControl"));
        ASPxGridView.ClientSideEvents.ToolbarItemClick = "function(s, e) { if (e.item.name == 'ExportAsSeen') { e.processOnServer = true; e.usePostBack = true; } else if (e.item.name == 'saveLayouGridViewToolbarItem') { s.PerformCallback('save'); } }";
        ASPxGridView.ClientSideEvents.EndCallback = "function(s, e) { if (s.cpLayout != null) { " + popupControlClienInstanceName + ".Show(); " + popupASPxGriViewClientInstanceName + 
                                                    ".AddNewRow(); " + popupASPxGriViewClientInstanceName + ".StartEditRow(-1); setTimeout(function() { " + popupASPxGriViewClientInstanceName + 
                                                    ".SetEditValue('Layout', s.cpLayout); s.cpLayout = null; }, 300); } }";
        ASPxPopupControl.ClientSideEvents.CloseUp = "function(s, e) { " + popupASPxGriViewClientInstanceName + ".CancelEdit(); }";
    }
    else
        RecreateTemplatedItems(ASPxGridView.Toolbars.FindByName("gridViewToolbar"), "ASPxPopupControl");
     (ASPxPopupControl.FindControl("loadButton") as ASPxButton).ClientSideEvents.Click = 
        "function(s, e) { var focusedRowIndex = " + popupASPxGriViewClientInstanceName + ".GetFocusedRowIndex(); if (focusedRowIndex > -1) " + 
        popupASPxGriViewClientInstanceName + ".GetRowValues(focusedRowIndex, 'Layout', function(rowValues) { " + popupControlClienInstanceName + ".Hide(); " + 
        ASPxGridView.ClientInstanceName + ".PerformCallback('load_' + rowValues) }); }";
    (ASPxPopupControl.FindControl("cancelButton") as ASPxButton).ClientSideEvents.Click = "function(s, e) { " + popupControlClienInstanceName + ".Hide() }";
     ASPxGridView.ToolbarItemClick += ASPxGridView_ToolbarItemClick;
}

                                  Note that the process of switching ASPxGridView to the edit mode takes some time, so to set a value for the edited Layout cell, it is necessary to call the ASPxClientGridView.SetEditValue method with the same delay by using the setTimeout JavaScript function (in my sample, this delay is set to 300 milliseconds).

                                  - Also, add the ASPxGridView.CustomCallback event handler to load/restore ASPxGridView's layout on the server side:

                                  [C#]

    ...
    ASPxGridView.CustomCallback += ASPxGridView_CustomCallback;

private static void ASPxGridView_CustomCallback(object sender, ASPxGridViewCustomCallbackEventArgs e)
{
    ASPxGridView ASPxGridView = sender as ASPxGridView;
    string[] parameters = e.Parameters.Split('_');
     if (parameters.Length > 1)
    {
        if (parameters[0].Equals("load"))
            ASPxGridView.LoadClientLayout(parameters[1]);
    }
    else
    {
        if (e.Parameters.Equals("save"))
            ASPxGridView.JSProperties["cpLayout"] = ASPxGridView.SaveClientLayout();
    }
}

                                  As an alternative, it is possible to create an ASPxPopupControl at runtime and add it to the page's hierarchy. For this, create a separate ASPxPopupControlCreater class, in which define the CreateASPxPopupControl method for creating the ASPxPopupControl:

                                  [C#]
                                  public static ASPxPopupControl CreateASPxPopupControl()
{
    ASPxPopupControl ASPxPopupControl = new ASPxPopupControl();

    ASPxPopupControl.ID = "ASPxPopupControl";
    ASPxPopupControl.ClientInstanceName = "ASPxPopupControl";

    ASPxPopupControl.Controls.Add(CreateSqlDataSource());
    ASPxPopupControl.Controls.Add(CreateASPxGridView());

    ASPxButton loadButton = new ASPxButton();

    loadButton.ID = "loadButton";
    loadButton.Text = "Load";
    loadButton.AutoPostBack = false;

    ASPxPopupControl.Controls.Add(loadButton);

    ASPxButton cancelButton = new ASPxButton();

    cancelButton.ID = "cancelButton";
    cancelButton.Text = "Cancel";
    cancelButton.AutoPostBack = false;

    ASPxPopupControl.Controls.Add(cancelButton);

    return ASPxPopupControl;
}
                                  [C#]
                                  private static SqlDataSource CreateSqlDataSource()
{
    SqlDataSource sqlDataSource = new SqlDataSource();

    sqlDataSource.ID = "sqlDataSource";
    sqlDataSource.ConnectionString = ConfigurationManager.ConnectionStrings["connectionString"].ConnectionString;
    sqlDataSource.SelectCommand = "SELECT * FROM GridLayouts WHERE @GridID=GridID";
    sqlDataSource.SelectParameters.Add(new Parameter("GridID"));
    sqlDataSource.InsertCommand = "INSERT INTO GridLayouts VALUES (@Bookmark, @Description, @Layout, @GridID)";
    sqlDataSource.InsertParameters.Add(new Parameter("Bookmark"));
    sqlDataSource.InsertParameters.Add(new Parameter("Description"));
    sqlDataSource.InsertParameters.Add(new Parameter("Layout"));
    sqlDataSource.InsertParameters.Add(new Parameter("GridID"));
    sqlDataSource.DeleteCommand = "DELETE FROM GridLayouts WHERE ID = @ID";
    sqlDataSource.DeleteParameters.Add(new Parameter("ID"));
    sqlDataSource.UpdateCommand = "UPDATE GridLayouts SET Bookmark = @Bookmark, Description = @Description WHERE ID = @ID";
    sqlDataSource.UpdateParameters.Add(new Parameter("Bookmark"));
    sqlDataSource.UpdateParameters.Add(new Parameter("Description"));

    return sqlDataSource;
}
                                  [C#]
                                  private static ASPxGridView CreateASPxGridView()
{
    ASPxGridView ASPxGridView = new ASPxGridView();

    ASPxGridView.ID = "popupASPxGridView";
    ASPxGridView.ClientInstanceName = "popupASPxGridView";
    ASPxGridView.KeyFieldName = "ID";
    ASPxGridView.DataSourceID = "sqlDataSource";

    GridViewCommandColumn gridViewCommandColumn = new GridViewCommandColumn();

    gridViewCommandColumn.ShowEditButton = true;
    gridViewCommandColumn.ShowDeleteButton = true;

    ASPxGridView.Columns.Add(gridViewCommandColumn);

    GridViewDataColumn idGridViewDataColumn = new GridViewDataColumn();

    idGridViewDataColumn.FieldName = "ID";
    idGridViewDataColumn.Visible = false;

    ASPxGridView.Columns.Add(idGridViewDataColumn);
    ASPxGridView.Columns.Add(new GridViewDataColumn("Bookmark"));
    ASPxGridView.Columns.Add(new GridViewDataColumn("Description"));

    GridViewDataColumn layoutGridViewDataColumn = new GridViewDataColumn();

    layoutGridViewDataColumn.FieldName = "Layout";
    layoutGridViewDataColumn.ReadOnly = true;
    layoutGridViewDataColumn.Visible = false;
    layoutGridViewDataColumn.EditFormSettings.Visible = DefaultBoolean.True;

    ASPxGridView.Columns.Add(layoutGridViewDataColumn);

    GridViewDataColumn gridIDGridViewDataColumn = new GridViewDataColumn();

    gridIDGridViewDataColumn.FieldName = "GridID";
    gridIDGridViewDataColumn.ReadOnly = true;
    gridIDGridViewDataColumn.Visible = false;
    gridIDGridViewDataColumn.EditFormSettings.Visible = DefaultBoolean.True;

    ASPxGridView.Columns.Add(gridIDGridViewDataColumn);

    ASPxGridView.EditFormLayoutProperties.SettingsItems.Width = Unit.Pixel(200);
    ASPxGridView.SettingsBehavior.AllowFocusedRow = true;

    return ASPxGridView;
}

                                  private static SqlDataSource CreateSqlDataSource()
                                  {
                                      SqlDataSource sqlDataSource = new SqlDataSource();


                                      sqlDataSource.ID = "sqlDataSource";
                                      sqlDataSource.ConnectionString = ConfigurationManager.ConnectionStrings["connectionString"].ConnectionString;
                                      sqlDataSource.SelectCommand = "SELECT * FROM GridLayouts WHERE @GridID=GridID";
                                      sqlDataSource.SelectParameters.Add(new Parameter("GridID"));
                                      sqlDataSource.InsertCommand = "INSERT INTO GridLayouts VALUES (@Bookmark, @Description, @Layout, @GridID)";
                                      sqlDataSource.InsertParameters.Add(new Parameter("Bookmark"));
                                      sqlDataSource.InsertParameters.Add(new Parameter("Description"));
                                      sqlDataSource.InsertParameters.Add(new Parameter("Layout"));
                                      sqlDataSource.InsertParameters.Add(new Parameter("GridID"));
                                      sqlDataSource.DeleteCommand = "DELETE FROM GridLayouts WHERE ID = @ID";
                                      sqlDataSource.DeleteParameters.Add(new Parameter("ID"));
                                      sqlDataSource.UpdateCommand = "UPDATE GridLayouts SET Bookmark = @Bookmark, Description = @Description WHERE ID = @ID";
                                      sqlDataSource.UpdateParameters.Add(new Parameter("Bookmark"));
                                      sqlDataSource.UpdateParameters.Add(new Parameter("Description"));


                                      return sqlDataSource;
                                  }
