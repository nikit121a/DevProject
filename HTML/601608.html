                                Nikolai (DevExpress Support) 01.31.2018

                                Hi Victoria,

                                The calculateCustomSummary callback is intended for calculating summary values. If you wish to customize the cell markup, I recommend that you use the dxPivotGrid.onCellPrepared event handler for this task. The cellElement parameter represents a container of your cell. You can customize its content using the JavaScript or jQuery API.

                                • Victoria Ponchio 1 02.01.2018
                                  Hello Maxim,


                                  Here's the code we used to the ASPxPivotGrid :


                                  - On the HtmlCellPrepared event:


                                  [C#]
                                   settings.HtmlCellPrepared = (sender, e) => 
            {
                   _pivotDataSource = e.CreateDrillDownDataSource();
                  //in this example. i'm getting the value from the ["Codigo"] column
                    __codigoProjeto = _pivotDataSource[0] != null && _pivotDataSource[0]["Codigo"] != null ? string.Format("{0}", _pivotDataSource[0]["Codigo"].ToString()) : string.Empty;

                if (!_rowIsCollapsed && (e.RowValueType != PivotGridValueType.Total && e.RowValueType != PivotGridValueType.GrandTotal)) //we just setted the links on the last row levels
                {
                    if (e.DataField.SummaryType != PivotSummaryType.Custom)
                    {
                        _valor = 0;

                        if (!e.DataField.FieldName.ToLower().Contains("unbound")) //we called "unbound" these custom columns
                        {
                            _valor = RoundValue(Convert.ToDecimal(_pivotDataSource[0][e.DataField.FieldName]));
                        }

                       //this is just an custom format, please desconsider
                        var format = CasaDecimal > 0 ? "0:#,0." + "".PadRight(CasaDecimal, '0') : "0:#,0";
                        format = _valor % 2 != 0 && Divisor == 1000000 && CasaDecimal == 0 ? "0:#,0." + "".PadRight(1, '0') : format;

                        switch (e.DataField.FieldName.ToUpper()) //we searched the columns by their names to customize their texts
                        {
                            case "ORCADOAPROVADO": //the column name

                                e.Cell.Text = string.Format(UrlPopupDrillDown, _codigoProjeto, e.DataField.FieldName, string.Format("{" + format + "}", _valor), "#1F47A5", "bold", ""); //note that i'm in the "ORCADOAPROVADO" column, but i'm also using the _codigoProjeto value (that i got from the "Codigo" column) and that i keep the value "_valor", that is the value that were calculated in the "ORCADOAPROVADO" column

                                if (_valor < 0)
                                {
                                    e.Cell.Text = string.Format(UrlPopupDrillDown, _codigoProjeto, e.DataField.FieldName, string.Format("{" + format + "}", _valor), "#FF0000", "bold", "");
                            }
                                break;
                                }
                         }
                 }
            };

                                  - The UrlPopupDrillDown variable was defined like that:

                                  [C#]
                                      public string UrlPopupDrillDown
        {
            get //note that the {2} on the link format, is the '_valor' variable, that is the value calculated on the "ORCADOAPROVADO" column
            {
                return "<a onmouseover='this.style.textDecoration=&#39;underline&#39;' onmouseout='this.style.textDecoration=&#39;none&#39;'  href='javascript: masterplandetail.funcs.showPopupDrillDown(&#39;{0}&#39;, &#39;{1}&#39;, &#39;{2}&#39;, &#39;{5}&#39;);' style='text-decoration: none; color:{3}; font-weight:{4}'>{2}</a>";
            }

        }

                                  - The function that we setted on the links, the 'masterplandetail.funcs.showPopupDrillDown', called this method, that receive the informations we collected from another columns:

                                  [C#]
                                      public PartialViewResult LoadDrillDown(string codigoProjeto = "", string codigoComentario = "", string campo = "", string text = "", string extra = "") //we receive many variables because for each link we need a diferent value, in this case i'm using the 'codigoProjeto' variable
        {
            string url = string.Empty;

            ViewBag.UrlDrillDown = string.Empty;

            switch (campo.ToUpper())
            {
                case "ORCADOAPROVADO":
                    url = "http://10.250.250.250/ReportServer/Pages/ReportViewer.aspx?/REPORT/MANAGEMENT/PAGE_Orcamento&paramAnoMes=" + AnoMes + "&paramProjeto=" + codigoProjeto; //here we set the value that we got from the "Codigo" column (the codigoProjeto), ps. desconsider the variable 'AnoMes', this one we got from another place
                    break;
          }
       }
