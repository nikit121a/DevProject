                                  I was able to recreate the issue by adding a date edit item to the ribbon in the Init event. Maybe I'm customizing the toolbar at the wrong time. Change RichPartial.cshtml as follows and open in IE:

                                  [C#]
                                  // RichPartial.cshtml
@using DevExpress.Web.ASPxRichEdit
@Html.DevExpress().RichEdit(settings =>
{
    settings.Name = "RichEdit";
    settings.CallbackRouteValues = new { Controller = "Home", Action = "RichPartial" };
    settings.RibbonMode = RichEditRibbonMode.OneLineRibbon;
    settings.Init += (sender, args) =>
    {
        var edit = sender as MVCxRichEdit;
        edit.CreateDefaultRibbonTabs(true);
        var tab = edit.RibbonTabs.Find(p => p.Text == "Home");

        var item = new RibbonDateEditItem("Test Date");
        var g1 = tab.Groups[1];
        g1.Items.Insert(0, item);
    };
}).GetHtml()
